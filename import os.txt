import os
import requests
from dotenv import load_dotenv

load_dotenv()

key = os.getenv('AIRTABLE_API_KEY')
base = os.getenv('AIRTABLE_BASE_ID')

print(f"Key loaded: {key[:15]}..." if key else "KEY MISSING")
print(f"Base ID: {base}" if base else "BASE ID MISSING")

tables = [
    "Employment Contract Clauses",
    "Rewrite Strategies", 
    "PDPA Data Protection Clause",
    "Service Agreement"
]

for table in tables:
    url = f"https://api.airtable.com/v0/{base}/{table.replace(' ', '%20')}?maxRecords=1"
    r = requests.get(url, headers={"Authorization": f"Bearer {key}"})
    print(f"{table}: {r.status_code} ({'OK' if r.status_code == 200 else 'FAIL'})")
